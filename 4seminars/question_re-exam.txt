#include <stdio.h>

struct Base { 
    virtual int f() = 0;
};

struct Derived : Base {
    int f();
};

int Derived::f() {
    printf("a\n");
    return 0;
}

int main() {
  Derived d; 
  Base *b = &d;
  dynamic_cast<Derived*>(b); // Mush be invalid
}
